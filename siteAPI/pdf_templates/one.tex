\newcommand\textline[4][t]{%
  \par\smallskip\noindent
  \parbox[#1]{.55\textwidth}{\left{#3}}%
  \parbox[#1]{.20\textwidth}{\left{#2}}%
  \parbox[#1]{.20\textwidth}{\right{#4}}\par\smallskip%
  \vspace*{5pt}
}
\newcommand\sectiontitle[2][t]{%
	\vspace*{-4pt}%
	\noindent\makebox[\linewidth]{\rule{\textwidth}{0.1pt}}\\%
	\hspace*{2mm}\MakeUppercase{#2}\\%
	\noindent\makebox[\linewidth]{\rule[8pt]{\textwidth}{0.1pt}}%
	\vspace*{-7pt}
}
\newcommand\makeprofile[3][t]{%
	\sectiontitle[t]{profile}
	\hspace*{10mm}{Objective: \parbox[t][][t]{0.8\textwidth}{#2}\\}
	\ifthenelse{\equal{#3}{}}{}{\hspace*{10mm}{Availibility: #3}\\}%
}
